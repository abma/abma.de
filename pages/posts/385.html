Pearpc


wie bekomme ich netzwerk an den start?


ppc.conf
pci_rtl8139_installed = 1
pci_rtl8139_mac = "de:ad:ca:fe:12:35"
scripts/ifppc_up
echo UID=$UID
PPC_NETMASK="255.255.255.0"
PPC_HOST_IP="192.168.10.80"
PPC_CLIENT_IP="192.168.10.1"
/sbin/ifconfig ppc $PPC_HOST_IP netmask $PPC_NETMASK up &&\
/sbin/iptables -t nat -s $PPC_CLIENT_IP -A POSTROUTING -j MASQUERADE &&\
echo 1 > /proc/sys/net/ipv4/ip_forward
und dann noch macosx auf die ip 192.168.10.1 konfigurieren! (dann noch den echten dns-server/proxy-server eintragen)
