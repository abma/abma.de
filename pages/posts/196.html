Um Text auf der Konsole farbig zu machen eignet sich folgendes einfache Script:

#!/bin/sh
red=$(tput setaf 1) normal=$(tput sgr0)
WORD="suchwort"
cat protokolldatei.log |sed "s/${WORD}/${red}${WORD}${normal}/g"
