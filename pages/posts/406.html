(Autoconfiguration file for Proxy Settings in Netscape/Firefox/Internet Explorer/...)
function FindProxyForURL(url, host)
{
if (isInNet(host, "192.168.0.0","255.255.255.0") || (isInNet(host,"127.0.0.1",0))){
return "DIRECT";
} else {
return "PROXY proxy.lan:8080;"
}
}
